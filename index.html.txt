</div>

<script>
function entrar() {
    let nombre = document.getElementById("nombre").value;
    let pass = document.getElementById("pass").value;

    if(pass === "contraseña") {
        localStorage.setItem("nombre", nombre);

        document.getElementById("login").classList.add("hidden");
        document.getElementById("continuar").classList.remove("hidden");
        document.getElementById("user").innerText = nombre;
    } else {
        alert("Contraseña incorrecta 😅");
    }
}

function mostrarPregunta() {
    document.getElementById("continuar").classList.add("hidden");
    document.getElementById("pregunta").classList.remove("hidden");
}

function respuestaSi() {
    let nombre = localStorage.getItem("nombre");

    localStorage.setItem("respuesta", "Sí 💖");

    document.getElementById("pregunta").classList.add("hidden");
    document.getElementById("no").classList.add("hidden");
    document.getElementById("si").classList.remove("hidden");

    lanzarCorazones();
}

function respuestaNo() {
    localStorage.setItem("respuesta", "Intentó decir que no 😳");

    document.getElementById("pregunta").classList.add("hidden");
    document.getElementById("no").classList.remove("hidden");
}

function respuestaFinalNo() {
    localStorage.setItem("respuesta", "No ❌");

    alert("😅 Bueno… al menos lo intenté");

    verResultado();
}

function moverBoton() {
    let btn = document.getElementById("btnNo");
    btn.style.position = "absolute";
    btn.style.top = Math.random() * window.innerHeight + "px";
    btn.style.left = Math.random() * window.innerWidth + "px";
}

function lanzarCorazones() {
    for(let i=0; i<30; i++){
        let heart = document.createElement("div");
        heart.innerHTML = "💖";
        heart.className = "heart";
        heart.style.left = Math.random()*100 + "vw";
        heart.style.fontSize = (Math.random()*20 + 20) + "px";
        document.body.appendChild(heart);

        setTimeout(() => heart.remove(), 5000);
    }
}

function verResultado() {
    let nombre = localStorage.getItem("nombre");
    let respuesta = localStorage.getItem("respuesta");

    document.getElementById("si").classList.add("hidden");
    document.getElementById("no").classList.add("hidden");
    document.getElementById("resultado").classList.remove("hidden");

    document.getElementById("textoResultado").innerText =
        nombre + " respondió: " + respuesta;
}
</script>
